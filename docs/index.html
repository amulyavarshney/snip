class EmailValidator {
  constructor(private pattern = EMAIL_RE) {}

  validate(email: string): boolean {
    if (typeof email !== 'string') throw new TypeError('must be string');
    email = email.trim();
    if (!email) return false;
    return this.pattern.test(email);
  }
}

function validateEmail(email: string) {
  return new EmailValidator().validate(email);  // convenience wrapper
}

Team config

Commit a .snip.json to your repo. Every developer and CI run picks it up automatically. One engineer sets the floor, the whole team keeps it. 

{
  "mode": "full",
  "language": "auto",
  "prod": { "protect": ["auth", "payments", "billing"] },
  "score": { "minScore": 60, "failBelow": true }
}

Add to CI to enforce the score floor on every PR:

# .github/workflows/snip.yml
- run: npx @amulyavarshney/snip score src/ --min-score 60 --fail-below
